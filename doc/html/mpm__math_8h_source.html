41 /*
42  * Polar Decomposition M = R * T
43  * where R is a rotation and T a pure translation/stretch matrix.
44  *
45  * The decomposition is achieved using SVD, i.e. M = U S V^T,
46  * where U = R V and S is diagonal.
47  *
48  *
49  * For any physically admissible deformation gradient, the determinant of R must equal +1.
50  * However, scenerios can arise, where the particles interpenetrate and cause inversion, leading to a determinant of R equal to -1.
51  * In this case, the inversion direction is heuristically identified with the eigenvector of the smallest entry of S, which should work for most cases.
52  * The sign of this corresponding eigenvalue is flipped, the original matrix M is recomputed using the flipped S, and the rotation and translation matrices are
53  * obtained again from an SVD. The rotation should proper now, i.e., det(R) = +1.
54  */
55 
56 static inline bool PolDec(Matrix3d M, Matrix3d &R, Matrix3d &T, bool scaleF) {
57 
58  JacobiSVD<Matrix3d> svd(M, ComputeFullU | ComputeFullV); // SVD(A) = U S V*
59  Vector3d S_eigenvalues = svd.singularValues();
60  Matrix3d S = svd.singularValues().asDiagonal();
61  Matrix3d U = svd.matrixU();
62  Matrix3d V = svd.matrixV();
63  Matrix3d eye;
64  eye.setIdentity();
65 
66  // now do polar decomposition into M = R * T, where R is rotation
67  // and T is translation matrix
68  R = U * V.transpose();
69  T = V * S * V.transpose();
70 
71  if (R.determinant() < 0.0) { // this is an improper rotation
72  // identify the smallest entry in S and flip its sign
73  int imin;
74  S_eigenvalues.minCoeff(&imin);
75  S(imin, imin) *= -1.0;
76 
77  R = M * V * S.inverse() * V.transpose(); // recompute R using flipped stretch eigenvalues
78  }
79 
80  /*
81  * scale S to avoid small principal strains
82  */
83 
84  if (scaleF) {
85  double min = 0.3; // 0.3^2 = 0.09, should suffice for most problems
86  double max = 2.0;
87  for (int i = 0; i < 3; i++) {
88  if (S(i, i) < min) {
89  S(i, i) = min;
90  } else if (S(i, i) > max) {
91  S(i, i) = max;
92  }
93  }
94  T = V * S * V.transpose();
95  }
96 
97  if (R.determinant() > 0.0) {
98  return true;
99  } else {
100  return false;
101  }
102 }

232 static inline Matrix3d LimitEigenvalues(Matrix3d S, double limitEigenvalue) {
233 
234  /*
235  * compute Eigenvalues of matrix S
236  */
237  SelfAdjointEigenSolver<Matrix3d> es;
238  es.compute(S);
239 
240  double max_eigenvalue = es.eigenvalues().maxCoeff();
241  double min_eigenvalue = es.eigenvalues().minCoeff();
242  double amax_eigenvalue = fabs(max_eigenvalue);
243  double amin_eigenvalue = fabs(min_eigenvalue);
244 
245  if ((amax_eigenvalue > limitEigenvalue) || (amin_eigenvalue > limitEigenvalue)) {
246  if (amax_eigenvalue > amin_eigenvalue) { // need to scale with max_eigenvalue
247  double scale = amax_eigenvalue / limitEigenvalue;
248  Matrix3d V = es.eigenvectors();
249  Matrix3d S_diag = V.inverse() * S * V; // diagonalized input matrix
250  S_diag /= scale;
251  Matrix3d S_scaled = V * S_diag * V.inverse(); // undiagonalize matrix
252  return S_scaled;
253  } else { // need to scale using min_eigenvalue
254  double scale = amin_eigenvalue / limitEigenvalue;
255  Matrix3d V = es.eigenvectors();
256  Matrix3d S_diag = V.inverse() * S * V; // diagonalized input matrix
257  S_diag /= scale;
258  Matrix3d S_scaled = V * S_diag * V.inverse(); // undiagonalize matrix
259  return S_scaled;
260  }
261  } else { // limiting does not apply
262  return S;
263  }

329 static inline char ComputeDirection(double xi, double yi, double xj, double yj, double xk, double yk) {

373 /* circle_circle_intersection() *
374  * Determine the points where 2 circles in a common plane intersect.
375  *
376  * int circle_circle_intersection(
377  * // center and radius of 1st circle
378  * double x0, double y0, double r0,
379  * // center and radius of 2nd circle
380  * double x1, double y1, double r1,
381  * // 1st intersection point
382  * double *xi, double *yi,
383  * // 2nd intersection point
384  * double *xi_prime, double *yi_prime)
385  *
386  * This is a public domain work. 3/26/2005 Tim Voght
387  *
388  */
389 //#include <stdio.h>

391 
392 static int circle_circle_intersection(double x0, double y0, double r0,
393  double x1, double y1, double r1,
394  double &xi, double &yi,
395  double &xi_prime, double &yi_prime)
396 {
397  double a, dx, dy, d, h, rx, ry;
398  double x2, y2;
399 
400  /* dx and dy are the vertical and horizontal distances between
401  * the circle centers.
402  */
403  dx = x1 - x0;
404  dy = y1 - y0;
405 
406  /* Determine the straight-line distance between the centers. */
407  //d = sqrt((dy*dy) + (dx*dx));
408  d = hypot(dx,dy); // Suggested by Keith Briggs
409 
410  /* Check for solvability. */
411  if (d > (r0 + r1))
412  {
413  /* no solution. circles do not intersect. */
414  return 0;
415  }
416  if (d < fabs(r0 - r1))
417  {
418  /* no solution. one circle is contained in the other */
419  return 0;
420  }
421 
422  /* 'point 2' is the point where the line through the circle
423  * intersection points crosses the line between the circle
424  * centers.
425  */
426 
427  /* Determine the distance from point 0 to point 2. */
428  a = ((r0*r0) - (r1*r1) + (d*d)) / (2.0 * d) ;
429 
430  /* Determine the coordinates of point 2. */
431  x2 = x0 + (dx * a/d);
432  y2 = y0 + (dy * a/d);
433 
434  /* Determine the distance from point 2 to either of the
435  * intersection points.
436  */
437  h = sqrt((r0*r0) - (a*a));
438 
439  /* Now determine the offsets of the intersection points from
440  * point 2.
441  */
442  rx = -dy * (h/d);
443  ry = dx * (h/d);
444 
445  /* Determine the absolute intersection points. */
446  xi = x2 + rx;
447  xi_prime = x2 - rx;
448  yi = y2 + ry;
449  yi_prime = y2 - ry;
450 
451  return 1;
452 }
